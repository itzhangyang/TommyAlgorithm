    private boolean bfs(int[] nums){
        int n = nums.length;
        boolean[] visited = new boolean[n];
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        visited[0] = true;

        while(!queue.isEmpty()){
            int curr = queue.poll();
            if(curr == n - 1) return true;

            int max = nums[curr];
            for(int i = 1; i <= max && curr + i < n;i++ ){
                int next = curr + i;
                if(!visited[next]){
                    queue.offer(next);
                    visited[next] = true;
                }
                
            }
        }

        return false;
    }
}
